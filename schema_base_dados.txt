CREATE TABLE DadosPessoais (
  matricula INTEGER UNSIGNED  NOT NULL   AUTO_INCREMENT,
  nome VARCHAR(100)  NULL  ,
  email VARCHAR(60)  NULL  ,
  celular VARCHAR(14)  NULL  ,
  data_nascimento DATE  NULL    ,
PRIMARY KEY(matricula));



CREATE TABLE usuarios (
  id INTEGER UNSIGNED  NOT NULL   AUTO_INCREMENT,
  nome VARCHAR(100)  NULL  ,
  endereco VARCHAR(200)  NULL  ,
  telefone VARCHAR(14)  NULL  ,
  email VARCHAR(60)  NULL  ,
  status INTEGER UNSIGNED  NULL    ,
PRIMARY KEY(id));



CREATE TABLE Profissao (
  estudante INTEGER UNSIGNED  NOT NULL  ,
  area_atuacao VARCHAR(60)  NULL  ,
  cargo VARCHAR(60)  NULL  ,
  regime_trabalho VARCHAR(60)  NULL  ,
  renda_mensal DECIMAL(10,2)  NULL    ,
INDEX profissao_FKIndex1(estudante),
  FOREIGN KEY(estudante)
    REFERENCES DadosPessoais(matricula)
      ON DELETE NO ACTION
      ON UPDATE NO ACTION);
